public class Operaciones{
  public static void main(String args[]){
  
    int numero_uno = 4;
    int numero_dos = 4;
    int resultado = 0;
    int resultado2 = 0;

    resultado = numero_uno + numero_dos / 2;
    resultado2 = numero_uno / 2 + numero_dos;
    
    System.out.println("Los resultados son: " + resultado + " y " + resultado2);
  }
}
